];


function weightedRandom(items){

  let total = items.reduce((sum,it)=>sum+it.weight,0);

  let r = Math.random() * total;

  let acc = 0;

  for(let it of items){

    acc += it.weight;

    if(r < acc) return it;

  }

}


function showRandom(){

  const it = weightedRandom(items);

  document.getElementById('heroImg').src = it.img;

  document.getElementById('quoteText').textContent = it.quote;

  document.getElementById('authorText').textContent = it.author;

}


// load ngẫu nhiên khi mở

showRandom();
